EXAMPLE 3: Monitor quantum metrics
──────────────────────────────────

const metrics = {
  bySpinState: {},
  totalSignals: 0,
  consensusCount: 0,
  conflictCount: 0,
  avgConfidence: 0
};

Object.values(window._predictions).forEach(pred => {
  if (!pred.kalshiExecution) return;
  
  const spinState = pred.kalshiExecution.spinState;
  metrics.bySpinState[spinState] = (metrics.bySpinState[spinState] || 0) + 1;
  metrics.totalSignals++;
  
  const alignment = pred.diagnostics?.quantumSpinState?.agreement?.alignmentScore || 0;
  if (alignment > 0.70) metrics.consensusCount++;
  else if (alignment > 0.3) metrics.conflictCount++;
  
  metrics.avgConfidence += pred.kalshiExecution.confidence;
});

metrics.avgConfidence = Math.round(metrics.avgConfidence / metrics.totalSignals);

console.log(metrics);
